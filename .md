package com.example.percabangan

import android.os.Bundle
import android.widget.Button
import android.widget.EditText
import android.widget.TextView
import androidx.activity.enableEdgeToEdge
import androidx.appcompat.app.AppCompatActivity
import androidx.core.view.ViewCompat
import androidx.core.view.WindowInsetsCompat

class MainActivity : AppCompatActivity() {
    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContentView(R.layout.activity_main)

        // Menghubungkan Komponen UI
        val edtInput = findViewById<EditText>(R.id.edtInput)
        val txtResult = findViewById<TextView>(R.id.txtResult)
        val btnIf = findViewById<Button>(R.id.btnIf)
        val btnIfElse = findViewById<Button>(R.id.btnIfElse)
        val btnIfElseIf = findViewById<Button>(R.id.btnIfElseIf)
        val btnWhen = findViewById<Button>(R.id.btnWhen)
        val btnWhenExpression = findViewById<Button>(R.id.btnWhenExpression)
        val btnWhenStatement = findViewById<Button>(R.id.btnWhenStatement)
        val btnWhenNoSubject = findViewById<Button>(R.id.btnWhenNoSubject)

        // Tombol IF
        btnIf.setOnClickListener {
            val input = edtInput.text.toString().toIntOrNull()
            if (input != null && input > 50) {
                txtResult.text = "IF: Angka lebih besar dari 50"
            } else {
                txtResult.text = "IF: Masukkan angka valid dan lebih besar dari 50"
            }
        }

        // Tombol IF ELSE
        btnIfElse.setOnClickListener {
            val input = edtInput.text.toString().toIntOrNull()
            if (input != null) {
                if (input >= 75) {
                    txtResult.text = "IF ELSE: Nilai Memuaskan"
                } else {
                    txtResult.text = "IF ELSE: Nilai Belum Memuaskan"
                }
            } else {
                txtResult.text = "IF ELSE: Masukkan angka valid"
            }
        }

        // Tombol IF ELSE-IF ELSE
        btnIfElseIf.setOnClickListener {
            val input = edtInput.text.toString().toIntOrNull()
            if (input != null) {
                val hasil = if (input >= 85) {
                    "A"
                } else if (input >= 70) {
                    "B"
                } else if (input >= 55) {
                    "C"
                } else {
                    "D"
                }
                txtResult.text = "IF ELSE-IF ELSE: Nilai = $hasil"
            } else {
                txtResult.text = "IF ELSE-IF ELSE: Masukkan angka valid"
            }
        }

        // Tombol WHEN
        btnWhen.setOnClickListener {
            val input = edtInput.text.toString().toIntOrNull()
            if (input != null) {
                val hari = when (input) {
                    1 -> "Minggu"
                    2 -> "Senin"
                    3 -> "Selasa"
                    4 -> "Rabu"
                    5 -> "Kamis"
                    6 -> "Jumat"
                    7 -> "Sabtu"
                    else -> "Bukan nama hari"
                }
                txtResult.text = "WHEN: $hari"
            } else {
                txtResult.text = "WHEN: Masukkan angka valid"
            }
        }

        // Tombol WHEN Expression
        btnWhenExpression.setOnClickListener {
            val input = edtInput.text.toString().toIntOrNull()
            if (input != null) {
                val hasil = when (input) {
                    1 -> "x == 1"
                    2 -> "x == 2"
                    else -> "x is neither 1 nor 2"
                }
                txtResult.text = "WHEN Expression: $hasil"
            } else {
                txtResult.text = "WHEN Expression: Masukkan angka valid"
            }
        }

        // Tombol WHEN Statement
        btnWhenStatement.setOnClickListener {
            val input = edtInput.text.toString().toIntOrNull()
            if (input != null) {
                when (input) {
                    1 -> txtResult.text = "WHEN Statement: x == 1"
                    2 -> txtResult.text = "WHEN Statement: x == 2"
                    else -> txtResult.text = "WHEN Statement: x is neither 1 nor 2"
                }
            } else {
                txtResult.text = "WHEN Statement: Masukkan angka valid"
            }
        }

        // Tombol Tanpa Subjek
        btnWhenNoSubject.setOnClickListener {
            val input = edtInput.text.toString().toIntOrNull()
            if (input != null) {
                when {
                    input > 100 -> txtResult.text = "WHEN No Subject: Angka lebih besar dari 100"
                    input in 50..100 -> txtResult.text = "WHEN No Subject: Angka antara 50-100"
                    else -> txtResult.text = "WHEN No Subject: Angka kurang dari 50"
                }
            } else {
                txtResult.text = "WHEN No Subject: Masukkan angka valid"
            }
        }
    }
}
